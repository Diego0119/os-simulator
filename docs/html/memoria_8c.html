else
{

memoria[i].tamano = 0;
memoria[i].estado = 0;
fprintf(stdout, "Proceso %d en EJECUCIÓN, bloque (%d)\n", proceso->pid, i);
ejecutar_proceso(memoria, proceso, i);
verificar_fragmentacion(memoria, i);
